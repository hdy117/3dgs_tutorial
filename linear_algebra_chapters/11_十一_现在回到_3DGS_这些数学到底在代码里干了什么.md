```python
import numpy as np

# ====== 场景设置 ======
# 高斯在世界空间的参数
mu_world = np.array([1.0, 2.0, 3.0])  # 中心位置
Sigma_world = np.array([
    [4.0, 1.5, 0.8],
    [1.5, 3.0, 1.2],
    [0.8, 1.2, 2.0]
])  # 协方差矩阵 (对称正定)

# 相机外参 (旋转 + 平移)
R = np.array([
    [0.99, -0.03, 0.14],   # 近似绕 y 轴旋转 8°
    [0.03, 0.998, -0.06],
    [-0.14, 0.06, 0.99]
])
t = np.array([-1.0, -2.0, -5.0])  # 相机位置

print("=" * 70)
print("🔥 Step 1: 世界空间 → 相机空间")
print("=" * 70)
print(f"\n世界空间中心:{mu_world}")
print(f"世界空间协方差:\n{Sigma_world}")

# Step 2: 位置变换
mu_cam = R @ mu_world + t
print(f"\nStep 1a: 相机空间中心:")
print(f"μ_cam = R @ μ_world + t")
print(f"{R} @ {mu_world} + {t}")
print(f"= {mu_cam}")

# Step 3: 协方差传播 (关键点:Σ_cam = R Σ_world R^T)
Sigma_cam = R @ Sigma_world @ R.T
print(f"\nStep 1b: 相机空间协方差:")
print(f"Σ_cam = R @ Σ_world @ R.T")
print(f"\n✅ 对称吗?{np.allclose(Sigma_cam, Sigma_cam.T)}")

# Step 4: 验证特征分解的正确性(主轴方向被正确旋转)
eigenvalues_world, eigenvectors_world = np.linalg.eigh(Sigma_world)
eigenvalues_cam, eigenvectors_cam = np.linalg.eigh(Sigma_cam)

print(f"\n世界空间特征值:{eigenvalues_world}")
print(f"相机空间特征值:{eigenvalues_cam}")
print(f"✅ 特征值相同吗?(只旋转不缩放) {np.allclose(eigenvalues_world, eigenvalues_cam)}")

# Step 5: 验证主轴方向被正确旋转
print(f"\n世界空间主轴 v1 = {eigenvectors_world[:, 0]}")
print(f"相机空间主轴 u1 = {eigenvectors_cam[:, 0]}")
print(f"验证:R @ v1 ≈ u1 ?")
print(f"      R @ v1 = {R @ eigenvectors_world[:, 0]}")
print(f"✅ {'正确!' if np.allclose(R @ eigenvectors_world[:, 0], eigenvectors_cam[:, 0]) else '错误!'}")

print("\n🎯 关键理解:")
print("   - 位置用仿射变换:μ_cam = R μ_world + t")
print("   - 协方差用二次型传播:Σ_cam = R Σ_world R.T (旋转椭球)")
print("   - 特征值不变,只是主轴方向被旋转到相机基!")

# ====== Step 2: 投影到屏幕空间 ======
print("\n" + "=" * 70)
print("🔥 Step 2: 相机空间 → 屏幕空间 (透视投影)")
print("=" * 70)

# 相机内参
fx, fy = 800.0, 800.0  # 焦距
cx, cy = 640.0, 360.0  # 主点

X, Y, Z = mu_cam
print(f"\n相机空间中心:({X:.2f}, {Y:.2f}, {Z:.2f})")
print(f"相机内参:fx={fx}, fy={fy}, cx={cx}, cy={cy}")

# 投影到归一化设备坐标 (NDC) → 像素坐标
u_pixel = fx * X / Z + cx
v_pixel = fy * Y / Z + cy
print(f"\n屏幕中心:({u_pixel:.1f}, {v_pixel:.1f})")

# Step: 计算 Jacobian J = ∂(proj)/∂x
J = np.array([
    [fx / Z,     0.0,   -fx * X / (Z**2)],
    [0.0,       fy / Z, -fy * Y / (Z**2)]
])

print(f"\n投影 Jacobian J ({J.shape}):")
print(J)
print("\n每个元素的意义:")
print("  J[0,0] = fx/Z  → x 方向缩放因子")
print("  J[0,2] = -fx*X/Z2 → 深度变化对 u 的影响 (透视效应)")

# Step: 协方差传播到屏幕空间 Σ_2d ≈ J @ Σ_cam @ J.T
Sigma_cam_2x2 = Sigma_cam[:2, :2]  # 取 x-y 平面部分(简化)
Sigma_2d = J[:, :2] @ Sigma_cam_2x2 @ J[:, :2].T

print(f"\n屏幕空间协方差 Σ_2d:")
print(Sigma_2d)
print(f"✅ 对称吗?{np.allclose(Sigma_2d, Sigma_2d.T)}")

# Step: 提取椭圆的视觉特征
eigenvalues_2d, eigenvectors_2d = np.linalg.eigh(Sigma_2d)
axis_lengths = np.sqrt(eigenvalues_2d)
angle_deg = np.degrees(np.arctan2(eigenvectors_2d[1, 0], eigenvectors_2d[0, 0]))

print(f"\n椭圆特征:")
print(f"  半轴长度:{axis_lengths:.3f} pixels")
print(f"  主轴角度:{angle_deg:.1f}° (从 x 轴逆时针)")
```

## 🔥 第二部分：调试案例库——常见的线性代数错误

在实际工程中，协方差矩阵和 SVD 相关的 bug 经常出现在这些地方:

### 🐛 Bug #1: 协方差不是对称的！

**症状**: `np.linalg.eigh()`报错或得到奇怪结果。

**原因**: 数值误差或计算顺序错误导致 $Σ \neq Σ^T$。

```python
import numpy as np

def fix_asymmetric_covariance(Sigma):
    """强制协方差对称化"""
    
    # 方法 1: 直接平均 (最常用)
    Sigma_sym = (Sigma + Sigma.T) / 2
    
    # 方法 2: 如果知道原始来源，重新计算 Σ = X^T X / N
    
    return Sigma_sym

# ====== 测试 ======
np.random.seed(42)
X = np.random.randn(100, 3)
Sigma_raw = X.T @ X / len(X)

print("原始协方差 (理论上对称):")
max_asymm = np.max(np.abs(Sigma_raw - Sigma_raw.T))
print(f"最大不对称度：{max_asymm:.2e}")

# 人为制造一点噪声
Sigma_noisy = Sigma_raw + np.random.randn(3, 3) * 1e-10
print("\n加入数值噪声后:")
print(f"最大不对称度：{np.max(np.abs(Sigma_noisy - Sigma_noisy.T)):.2e}")

# 修复
Sigma_fixed = fix_asymmetric_covariance(Sigma_noisy)
print("修复后:")
print(f"最大不对称度：{np.max(np.abs(Sigma_fixed - Sigma_fixed.T)):.2e}")
```

### 🐛 Bug #2: 特征值为负数（协方差半正定被破坏）

**症状**: `sqrt(eigenvalues)`报错 "negative value cannot be raised to power"。

**原因**: 
1. 数值误差导致微小负数 (如 -1e-15)
2. 真正的非半正定矩阵（计算错误）

```python
import numpy as np

def safe_sqrt_eigenvalues(Sigma):
    """安全地提取标准差，处理数值误差"""
    
    # Step 1: 强制对称化
    Sigma = (Sigma + Sigma.T) / 2
    
    # Step 2: 特征分解
    eigenvalues = np.linalg.eigvalsh(Sigma)
    
    # Step 3: 裁剪负数到 0
    eigenvalues_safe = np.maximum(eigenvalues, 1e-10)  # 设置下限
    
    # Step 4: 开方得到标准差
    stds = np.sqrt(eigenvalues_safe)
    
    return eigenvalues, eigenvalues_safe, stds


# ====== 测试 ======
Sigma_problematic = np.array([
    [1.0, 0.5],
    [0.5, 1.0]
])

eigs_raw, eigs_safe, stds = safe_sqrt_eigenvalues(Sigma_problematic)

print(f"原始特征值：{eigs_raw}")
print(f"安全特征值 (裁剪后): {eigs_safe}")
print(f"标准差：{stds}")

# ====== 检测严重问题 ======
if np.any(eigs_raw < -1e-8):
    print("\n⚠️ 警告：存在显著的负特征值！")
    print("这可能是计算错误导致的非半正定协方差!")
```

### 🐛 Bug #3: Jacobian 计算错误导致投影失败

**症状**: 渲染结果有黑色斑点或高斯"消失"。

**原因**: 
1. Z=0附近除以零
2. Jacobian 行列式为 0（秩亏）

```python
import numpy as np

def safe_project_covariance(Sigma_cam, X, Y, Z, fx, fy):
    """安全的协方差投影，处理边界情况"""
    
    # 检查深度是否有效
    if Z <= 1e-6:
        print("⚠️ 警告：Z 过小，可能导致数值不稳定!")
        return None
    
    # 计算 Jacobian
    J = np.array([
        [fx / Z,     0.0,   -fx * X / (Z**2)],
        [0.0,       fy / Z, -fy * Y / (Z**2)]
    ])
    
    # 取协方差的前 2x2 部分 (XY 平面)
    Sigma_cam_2x2 = Sigma_cam[:2, :2]
    
    # 投影
    Sigma_2d = J[:, :2] @ Sigma_cam_2x2 @ J[:, :2].T
    
    # 检查 Jacobian 的秩
    rank_J = np.linalg.matrix_rank(J)
    if rank_J < 2:
        print("⚠️ 警告：Jacobian 秩亏，投影可能失败!")
    
    return Sigma_2d


# ====== 测试 ======
Sigma_cam = np.array([
    [0.02, 0.00],
    [0.00, 0.01]
])

X, Y, Z = 0.3, 0.1, 2.0
fx, fy = 800.0, 800.0

Sigma_2d = safe_project_covariance(Sigma_cam, X, Y, Z, fx, fy)
print(f"投影后的 Σ_2d:\n{Sigma_2d}")
```

### 🐛 Bug #4: 旋转顺序错误（RΣR^T vs R^TΣR）

**症状**: 高斯的朝向不对，椭球"歪了"。

**原因**: 混淆了主动变换和被动变换。

```python
import numpy as np

def verify_rotation_direction(Sigma, R):
    """验证协方差旋转的方向是否正确"""
    
    # 正确的传播公式（从世界到相机）:
    Sigma_cam = R @ Sigma @ R.T
    
    # 错误的方式：Σ' = R^T Σ R (这会反向旋转!)
    Sigma_wrong = R.T @ Sigma @ R
    
    print("验证旋转方向:")
    
    # 取一个特征向量 v，看 R @ v 是否等于新的特征向量
    eigenvalues, eigenvectors = np.linalg.eigh(Sigma)
    v1 = eigenvectors[:, 0]
    
    Sigma_cam_eigs, Sigma_cam_evecs = np.linalg.eigh(Sigma_cam)
    u1 = Sigma_cam_evecs[:, 0]
    
    print(f"世界空间主轴：v₁ = {v1}")
    print(f"相机空间主轴：u₁ = {u1}")
    print(f"验证：R @ v₁ ≈ u₁ ?")
    print(f"      R @ v₁ = {R @ v1}")
    
    if np.allclose(R @ v1, u1):
        print("✅ 正确！Σ_cam = R Σ R.T")
    else:
        print("❌ 错误！可能用了错误的公式")

# ====== 测试 ======
Sigma_world = np.array([[4.0, 1.5], [1.5, 2.0]])
R_test = np.array([
    [np.cos(np.pi/6), -np.sin(np.pi/6)],
    [np.sin(np.pi/6),  np.cos(np.pi/6)]
```

### 🐛 Bug #5: SVD 的 U,V符号不确定性导致问题

**症状**: 在某些情况下，特征分解结果"跳变"（符号反转）。

**原因**: 奇异值分解中 $u_i$和$v_i$可以同时取反而不影响 $A = UΣV^T$。

```python
import numpy as np

def check_svd_sign_consistency(A):
    """检查 SVD 的符号一致性"""
    
    U, S, Vt = np.linalg.svd(A)
    
    print("SVD 结果:")
    print(f"U:\n{U}")
    print(f"S: {S}")
    print(f"Vᵀ:\n{Vt}")
    
    # 验证核心关系 A @ v_i = σ_i * u_i
    print("\n验证关系 A @ vᵢ = σᵢ * uᵢ:")
    for i in range(len(S)):
        vi = Vt[i]
        ui = U[:, i]
        
        left = A @ vi
        right = S[i] * ui
        
        consistent = np.allclose(left, right) or np.allclose(left, -right)
        print(f"  i={i+1}: {'✅' if consistent else '❌'}")

# ====== 测试 ======
A_test = np.array([[3.0, 1.0], [1.0, 2.0]])
check_svd_sign_consistency(A_test)
```

## 🔥 第三部分：与第 8、9 章的直接对比

### 📊 SVD vs 特征分解在 3DGS 中的不同用途

| 场景 | 使用特征分解还是 SVD？ | 为什么？ | Python 实现 |
|------|--------------------|---------|-----------|
| **协方差矩阵分析** | ✅ 特征分解 (`eigh`) | Σ是对称正定 → $Σ=QΛQ^T$ | `eigenvalues, axes_dirs = np.linalg.eigh(Sigma)` |
| **Hessian 近似** | ⚠️ SVD | Hessian 可能不是对称的 | `U, S, Vt = np.linalg.svd(H)` |
| **低秩更新** | ✅ 截断 SVD | $A_k = \sum_{i=1}^k σ_i u_i v_i^T$ | `Ak = U[:,:k] @ diag(S[:k]) @ Vt[:k,:]` |
| **条件数估计** | ⚠️ SVD | $\kappa = σ_{max}/σ_{min}$ | `cond = S[0]/S[-1]` |

### 🎯 代码对比：什么时候用什么？

```python
import numpy as np

def choose_decomposition_method(Sigma, is_symmetric=True):
    """根据矩阵性质选择合适的分解方法"""
    
    if is_symmetric:
        # ✅ 特征分解 (更快、更稳定)
        eigenvalues, eigenvectors = np.linalg.eigh(Sigma)
        
        print("使用特征分解:")
        print(f"  公式：Σ = QΛQ^T")
        print(f"  特征值: {eigenvalues}")
        print(f"  主轴方向:\n{eigenvectors}")
        
    else:
        # ⚠️ SVD (通用但稍慢)
        U, S, Vt = np.linalg.svd(Sigma)
        
        print("\n使用 SVD:")
        print(f"  公式：A = UΣV^T")
        print(f"  奇异值: {S}")
        print(f"  U:\n{U}")
        print(f"  Vᵀ:\n{Vt}")
    
    # 验证结果
    if is_symmetric:
        reconstructed = eigenvectors @ np.diag(eigenvalues) @ eigenvectors.T
        print(f"\n✅ 重构误差：{np.linalg.norm(Sigma - reconstructed):.2e}")

# ====== 测试 ======
Sigma_sym = np.array([[4.0, 1.5], [1.5, 2.0]])
Sigma_non_sym = np.array([[3.0, 1.0], [0.5, 2.0]])

print("=" * 60)
print("对称矩阵:")
choose_decomposition_method(Sigma_sym, is_symmetric=True)

print("\n" + "=" * 60)
print("非对称矩阵:")
choose_decomposition_method(Sigma_non_sym, is_symmetric=False)
```

#### 2. **深度理解题**
为什么 `Sigma_2d = J @ Sigma_cam @ J.T`是正确的?从统计和几何两个角度解释。

<details>
<summary>👉 点击查看答案</summary>

**答案:**

**统计视角**:
- $\Sigma_{\text{cam}}$是 3D 高斯的协方差(描述在相机空间中的分布)
- $J = \partial p/\partial x$是从 3D 到 2D 的 Jacobian
- 对于线性变换$\mathbf{y}=A\mathbf{x}$,有 $\Sigma_y = A\Sigma_x A^T$
- 所以 $\Sigma_{2d} = J \Sigma_{\text{cam}} J^T$

**几何视角**:
- $\Sigma_{\text{cam}}$定义了一个椭球(主轴 + 尺度)
- $J$把这个椭球"投影"到屏幕平面
- 投影后的形状仍然是二次型,新协方差就是$J \Sigma_{\text{cam}} J^T$

**关键点**:
1. 透视投影本身是非线性的(有 1/Z)
2. 但在局部可以用 Jacobian 线性化近似
3. 一旦线性化,协方差传播公式就成立!

</details>

#### 3. **调试思考题**
假设你的 3DGS 渲染结果出现"奇怪的条纹"或"高斯消失",从线性代数角度分析可能的原因:

<details>
<summary>👉 点击查看答案(排查清单)</summary>

| 现象 | 可能原因 | 线性代数诊断 |
|------|---------|------------|
| **高斯被压扁成线** | $\det(\Sigma)\approx0$ | 至少一个特征值接近 0 |
| **渲染有黑色斑点** | Jacobian 计算错误 | $J$的某一行全为 0 |
| **旋转后方向不对** | $R\Sigma R^T$顺序错 | 应该是左乘和右乘转置 |
| **深度排序混乱** | 齐次坐标处理错 | w 分量没正确处理 |
| **边缘有锯齿** | Sigma_2d 计算错 | 局部线性化范围太大 |

**调试建议**:

#### 4. **代码实现题**
写一个函数,从协方差矩阵 $\Sigma$提取主轴方向和标准差。

<details>
<summary>👉 点击查看答案</summary>

```python
import numpy as np

def extract_gaussian_properties(Sigma):
    """
    从协方差矩阵提取高斯属性

    Args:
        Sigma: (3,3) 对称半正定矩阵

    Returns:
        directions: (3,3) 主轴方向,每列一个方向
        stds: (3,) 沿各主轴的标准差
    """
    # 特征分解(eigh 适用于对称矩阵)
    eigenvalues, eigenvectors = np.linalg.eigh(Sigma)

    # 按特征值从大到小排序
    idx = np.argsort(eigenvalues)[::-1]
    eigenvalues = eigenvalues[idx]
    eigenvectors = eigenvectors[:, idx]

    # 标准差(特征值的平方根)
    stds = np.sqrt(np.maximum(eigenvalues, 0))  # 防止数值误差导致负数

    return eigenvectors, stds

# 测试
Sigma = np.array([[4, 2, 0], [2, 5, 0], [0, 0, 3]])
directions, stds = extract_gaussian_properties(Sigma)
print("主轴方向:\n", directions)
print("标准差:", stds)
```

**关键点**:
- `eigh`比`eig`更适合对称矩阵(更快、更稳定)
- 特征值排序是为了让"最重要的主轴"排在前面
- 用 `max(0, ...)`防止数值误差导致负数开方

#### 5. **综合应用题**
给定一个高斯:
- 中心 $\boldsymbol{\mu} = [1, 2, 3]$(世界坐标)
- 协方差 $\Sigma_{\text{world}}$(已知)
- 相机外参 $R, t$

计算这个高斯在屏幕上的投影位置和 footprint。

<details>
<summary>👉 点击查看答案</summary>

**完整流程**:

```python
# Step 1: 世界→相机空间
mu_cam = R @ mu_world + t
Sigma_cam = R @ Sigma_world @ R.T

# Step 2: 投影到归一化设备坐标 (NDC)
X, Y, Z = mu_cam
u_ndc = fx * X / Z + cx  # 内参
v_ndc = fy * Y / Z + cy

# Step 3: 计算 Jacobian(在中心点处)
J = np.array([
    [fx/Z,     0,   -fx*X/(Z**2)],
    [0,       fy/Z, -fy*Y/(Z**2)]
])

# Step 4: 协方差传播到屏幕空间
Sigma_2d = J @ Sigma_cam[:3,:3] @ J.T  # 取前 3x3 部分(忽略 w)

# Step 5: 计算 alpha(二次型评估)
def gaussian_alpha(x_pixel, mu_2d, Sigma_2d):
    diff = x_pixel - mu_2d
    exponent = -0.5 * diff.T @ np.linalg.inv(Sigma_2d) @ diff
    return np.exp(exponent)

# 结果:
# - u_ndc, v_ndc: 投影中心(像素坐标)
# - Sigma_2d: 屏幕上的椭圆 footprint
# - gaussian_alpha(): 任意像素的权重
```

**关键公式总结**:
$$\begin{aligned} \mathbf{x}_{\text{cam}} &= R\mathbf{x}_{\text{world}} + t \\ \Sigma_{\text{cam}} &= R\Sigma_{\text{world}}R^T \\ \Sigma_{2d} &\approx J\Sigma_{\text{cam}}J^T \\ \alpha(\mathbf{p}) &= \exp(-0.5(\mathbf{p}-\boldsymbol{\mu}_{2d})^T\Sigma_{2d}^{-1}(\mathbf{p}-\boldsymbol{\mu}_{2d})) \end{aligned}$$

理论懂了,代码也看了。但还有一些容易混淆的概念需要澄清...

下一章会帮你梳理清楚!🔥

### 11.2 `x_cam = R @ x_world + t` 在干什么

它做了两件事:

1. 用 `R` 把方向和坐标轴对齐到相机基
2. 用 `t` 处理相机位置带来的平移

所以这是一个**仿射变换**,不是纯线性。

工程上它对应的就是:

- 世界空间的点,换到相机自己的参考系里
- 之后你才能谈投影、深度、遮挡

### 11.3 `Sigma_cam = R @ Sigma_world @ R^T` 在干什么

它在把世界空间里的椭球,正确旋转到相机空间里。

这不是单纯"把数字乘一下",而是在传播椭球的:

- 主轴方向
- 沿主轴的尺度

所以你可以把这一步读成:

> 把高斯的几何形状一起转到相机眼里去。

### 11.4 透视投影为什么不是线性的

如果相机坐标中的点是:

$$\mathbf{x}_{\text{cam}} = [X, Y, Z]^T$$

那图像坐标通常是:

$$\begin{aligned} u &= \frac{f_x X}{Z} + c_x \\ v &= \frac{f_y Y}{Z} + c_y \end{aligned}$$

这里有除法 `1 / Z`,所以它不是全局线性的。

这很重要,因为它解释了为什么:

- 世界到相机可以直接用矩阵处理
- 相机到屏幕不能直接靠一个固定矩阵精确搞定全部局部几何

### 11.5 `Sigma_2d ≈ J @ Sigma_cam @ J^T` 为什么成立

虽然透视投影全局不是线性的,但在某个高斯中心附近,你可以只看很小的局部扰动:

$$\delta \mathbf{p} \approx J \delta \mathbf{x}$$

这里 `J` 是投影函数在当前点处的 Jacobian。

对于上面的投影公式,`J` 可以写成一个 `2x3` 矩阵:

$$J = \begin{bmatrix} f_x/Z & 0 & -f_x X/Z^2 \\ 0 & f_y/Z & -f_y Y/Z^2 \end{bmatrix}$$

一旦局部近似成线性,协方差就能按线性规则传播:

$$\Sigma_{\text{2d}} \approx J \Sigma_{\text{cam}} J^T$$

这一步特别重要,因为它说明:

> 即使原问题是非线性的,只要局部足够小,工程师仍然会尽量把它拉回线性代数能处理的地盘。

### 11.6 `d^T @ Sigma^{-1} @ d` 在干什么

这里的 `d` 是像素点相对高斯中心的偏移:

$$\mathbf{d} = \mathbf{p} - \boldsymbol{\mu}_{\text{2d}}$$

如果只用普通欧氏距离,你默认每个方向都一视同仁。
但椭球不是球。它在不同方向上宽窄不同。

所以这里真正做的是:

> 按照椭球自己的主轴尺度,重新定义"离中心有多远"。

然后高斯权重就可以写成:

$$g(\mathbf{d}) = \exp\left(-\frac{1}{2} \mathbf{d}^T \Sigma_{\text{2d}}^{-1} \mathbf{d}\right)$$

这里的指数衰减就是 `2D` footprint 的核心。

### 11.7 `det(Sigma)`、`Sigma^{-1}`、`J @ Sigma @ J^T` 为什么老一起出现

在 3DGS 的高斯里,这几样东西其实是在描述同一个几何对象的不同侧面:

- `Sigma`:椭球本体
- 特征值/特征向量:椭球的主轴与尺度
- `det(Sigma)`:椭球的整体体积尺度
- `Sigma^{-1}`:按椭球几何定义的距离度量
- `J @ Sigma_cam @ J^T`:把椭球从 `3D` 局部传播到 `2D` 局部

你可以把它们理解成看同一个对象的不同镜头,而不是几套互不相干的公式。

### 11.8 一个最小 Python 例子:从 `3D` 协方差到 `2D` footprint

```python
import numpy as np

# 假设高斯中心在相机坐标里
X, Y, Z = 0.3, 0.1, 2.0
fx, fy = 800.0, 800.0

Sigma_cam = np.array([
    [0.02, 0.00, 0.00],
    [0.00, 0.01, 0.00],
    [0.00, 0.00, 0.03],
])

J = np.array([
    [fx / Z, 0.0, -fx * X / (Z**2)],
    [0.0, fy / Z, -fy * Y / (Z**2)],
])

Sigma_2d = J @ Sigma_cam @ J.T
print('Sigma_2d =\n', Sigma_2d)
print('eigenvalues =', np.linalg.eigvalsh(Sigma_2d))
```

你可以继续把 `Sigma_2d` 喂给前面"画协方差椭圆"的代码,就能直接看到屏幕上的 footprint。
